'use strict';

const mongoose = require('mongoose');
const Categories = require('./models-singular/categories');

const MONGOOSE_URI = 'mongodb://localhost/lab5';

mongoose.connect(MONGOOSE_URI, { useNewUrlParser: true });

const categories = new Categories();

/**
 *
 * @returns {Promise<void>}
 */
let testCategory = async () => {
  let testObject = {
    name: 'cars',
    description: 'brands of cars',
  };

  /**
   * Creates a new object in the mongo database using the category schema
   */
  let newCategory = await categories.create(testObject);
  console.log('Category Created', newCategory);

  /**
   * Gets all objects in the mongo database
   */
  let allCategories = await categories.get();
  console.log('All categories', allCategories);

  /**
   * Gets one object in the mongo database
   */
  let oneCategory = await categories.get('5d7ea6a8dfa67504408ea74e');
  console.log('One Category', oneCategory);

  /**
   * Updates an object in the mongo database using the category schema
   */
  let updateOne = await categories.update('5d7ea6a8dfa67504408ea74e',{name: 'airplanes', description: 'Cirrus'});
  console.log('Update one', updateOne);

  /**
   * Deletes one object in the mongo database
   */
  let deleteOne = await categories.delete('5d7ea3dd99bef50391d63fe5');
  console.log('Delete one', deleteOne);

  mongoose.disconnect();
  
};

testCategory();
